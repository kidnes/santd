<template>

        </div>
        <div style="margin-bottom: 16px">
            <s-input-search placeholder="input search text" enterButton="{{true}}" on-search="onSearch"></s-input-search>

export default {
    components: {

        console.log('the search value is: ', value);
    }
}
</script>
